# Types Reference

Complete TypeScript type definitions for the `@hare/agent` package.

## Agent Types

### HareAgentState

The state object persisted and synced with clients for `HareAgent`.

```ts
interface HareAgentState {
  /** Agent configuration ID */
  agentId: string
  /** Workspace ID */
  workspaceId: string
  /** Agent name */
  name: string
  /** Agent instructions/system prompt */
  instructions: string
  /** Model to use (e.g., '@cf/meta/llama-3.3-70b-instruct-fp8-fast') */
  model: string
  /** Conversation history */
  messages: ModelMessage[]
  /** Whether the agent is currently processing */
  isProcessing: boolean
  /** Last activity timestamp */
  lastActivity: number
  /** Connected user IDs */
  connectedUsers: string[]
  /** Scheduled tasks */
  scheduledTasks: ScheduledTask[]
  /** Agent status */
  status: 'idle' | 'processing' | 'error'
  /** Last error if any */
  lastError?: string
}
```

### McpAgentState

State for MCP (Model Context Protocol) agents.

```ts
interface McpAgentState {
  /** Workspace ID */
  workspaceId: string
  /** Number of connected clients */
  connectedClients: number
  /** Last activity timestamp */
  lastActivity: number
}
```

### ScheduledTask

Definition for scheduled one-time or recurring tasks.

```ts
interface ScheduledTask {
  /** Unique task ID */
  id: string
  /** Task type */
  type: 'one-time' | 'recurring'
  /** Execution timestamp (for one-time tasks) */
  executeAt?: number
  /** Cron expression (for recurring tasks) */
  cron?: string
  /** Action to perform */
  action: string
  /** Additional payload data */
  payload?: Record<string, unknown>
}
```

## Message Types

### ClientMessage

Messages sent from clients to agents.

```ts
interface ClientMessage {
  type: 'chat' | 'configure' | 'execute_tool' | 'get_state' | 'schedule'
  payload: unknown
}
```

### ServerMessage

Messages sent from agents to clients.

```ts
interface ServerMessage {
  type: 'text' | 'tool_call' | 'tool_result' | 'state_update' | 'error' | 'done'
  data: unknown
  timestamp: number
}
```

### ChatPayload

Payload for chat messages.

```ts
interface ChatPayload {
  /** The user's message */
  message: string
  /** User ID */
  userId: string
  /** Optional session ID */
  sessionId?: string
  /** Optional metadata */
  metadata?: Record<string, unknown>
}
```

### ToolExecutePayload

Payload for tool execution requests.

```ts
interface ToolExecutePayload {
  /** Tool identifier */
  toolId: string
  /** Tool parameters */
  params: Record<string, unknown>
}
```

### SchedulePayload

Payload for scheduling tasks.

```ts
interface SchedulePayload {
  /** Action to schedule */
  action: string
  /** When to execute (timestamp) */
  executeAt?: number
  /** Cron expression for recurring tasks */
  cron?: string
  /** Additional payload */
  payload?: Record<string, unknown>
}
```

## Tool Types

### Tool

Base tool definition.

```ts
interface Tool<TInput = unknown, TOutput = unknown> {
  /** Unique tool ID */
  id: string
  /** Tool description */
  description: string
  /** Zod schema for input validation */
  inputSchema: z.ZodSchema<TInput>
  /** Tool execution function */
  execute: (params: TInput, context: ToolContext) => Promise<ToolResult<TOutput>>
}
```

### ToolContext

Context provided to tool execution.

```ts
interface ToolContext {
  /** Cloudflare environment bindings */
  env: HareEnv
  /** Current workspace ID */
  workspaceId: string
  /** User ID executing the tool */
  userId: string
}
```

### ToolResult

Result returned from tool execution.

```ts
interface ToolResult<T = unknown> {
  /** Whether execution succeeded */
  success: boolean
  /** Result data (if successful) */
  data?: T
  /** Error message (if failed) */
  error?: string
}
```
